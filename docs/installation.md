# Manual
```bash
git clone https://git.yrwq.xyz/yrwq/yafetch && cd yafetch
make -j8
make CONF="${HOME}/.config/yafetch" config  # optional (copies default config to $HOME/.config/yafetch)
make strip -j8  # optional, strips the binary
sudo make install
```
